import torch
import yaml
from train import Trainer

with open('conf/config.yaml') as f:

trainer = Trainer(config)
ckpt = torch.load('checkpoints/best.pt', map_location='cpu', weights_only=False)
trainer.model.load_state_dict(ckpt['model_state_dict'])
trainer.model.to(trainer.device)

val_loss, metrics = trainer.validate(0)

print('\n=== POST-FIX BASELINE RESULTS (D-09) ===')
print(f'Total val loss: {val_loss:.4f}')
print(f'Gain MAE (matched): {metrics.get(\"gain\", 0):.2f} dB')
print(f'Gain MAE (raw):     {metrics.get(\"gain_raw\", 0):.2f} dB')
print(f'Freq MAE:           {metrics.get(\"freq\", 0):.3f} octaves')
print(f'Q MAE:              {metrics.get(\"q\", 0):.3f} decades')
print(f'Type accuracy:      {metrics.get(\"type_acc\", 0):.1%}')
print()
print('NOTE: This is the POST-fix baseline with corrected metrics and uniform data.')
print('Compare against pre_fix_baseline.md to see the delta (D-09).')
print('=== END POST-FIX BASELINE ===')
