import os
import random
import torch
from torch.utils.data import Dataset, DataLoader, random_split, Subset
import pytorch_lightning as pl
import numpy as np
from pathlib import Path
from model_cnn import normalize_params


class EQDataset(Dataset):
    """
    Dataset for loading synthetic EQ `.npz` files.
    Each file contains:
      'mel_stack' : float32 [3, 128, 431]  — [wet_mel, dry_mel, diff_mel] pre-computed in dB
      'label'     : float32 [24]           — 8 bands × [freq_norm, gain_norm, q_norm]
    """
    def __init__(self, data_dir: str | Path, training_mode: bool = False,
                 cache_in_ram: bool = False, global_norm: tuple | None = None):
        super().__init__()
        self.data_dir = Path(data_dir)
        self.training_mode = training_mode
        self.cache_in_ram = cache_in_ram
        self.files = sorted(self.data_dir.rglob("*.npz"))
        if not self.files:
            raise FileNotFoundError(f"No .npz files found in {self.data_dir}")

        # Global normalization stats: (mean[3], std[3]) per channel
        self._global_mean = None
        self._global_std = None
        if global_norm is not None:
            self._global_mean = torch.tensor(global_norm[0], dtype=torch.float32).view(3, 1, 1)
            self._global_std = torch.tensor(global_norm[1], dtype=torch.float32).view(3, 1, 1)

        if self.cache_in_ram:
            self._cache = [np.load(f) for f in self.files]

    def __len__(self) -> int:
        return len(self.files)

    def __getitem__(self, idx: int) -> tuple[torch.Tensor, torch.Tensor]:
        if self.cache_in_ram:
            data = self._cache[idx]
        else:
            data = np.load(self.files[idx])
        if "mel_stack" in data:
            x = torch.tensor(data["mel_stack"], dtype=torch.float32)
        else:
            # Fallback if keys are separate (common in generator output)
            wet = data["wet_mel"]
            dry = data["dry_mel"]
            diff = wet - dry
            x = torch.tensor(np.stack([wet, dry, diff], axis=0), dtype=torch.float32)

        label_key = "label" if "label" in data else "labels"
        y = torch.tensor(data[label_key], dtype=torch.float32)
        
        # Ensure labels are normalized into [0, 1] for loss calculation
        y = normalize_params(y)

        # Global normalization for input Mel-spectrogram
        # Preserves inter-channel relationships (wet vs dry vs diff) that
        # per-sample normalization would destroy.
        if self._global_mean is not None:
            x = (x - self._global_mean) / self._global_std.clamp(min=1e-6)

        # Training augmentation: frequency masking (SpecAugment-style)
        if self.training_mode:
            # Mask 1-3 random frequency bands (each up to 8 bins wide)
            n_mel_bins = x.shape[1]
            for _ in range(random.randint(1, 3)):
                f_start = random.randint(0, n_mel_bins - 1)
                f_width = random.randint(1, min(8, n_mel_bins - f_start))
                x[:, f_start:f_start + f_width, :] = 0.0

        return x, y


class EQDataModule(pl.LightningDataModule):
    """
    Lightning DataModule mapping the EQDataset into Train and Val loaders
    with standard 80/20 splits.
    """
    def __init__(self, data_dir: str | Path, batch_size: int = 128, num_workers: int = 8):
        super().__init__()
        self.data_dir = data_dir
        self.batch_size = batch_size
        self.num_workers = num_workers

        # These will be populated in setup()
        self.train_dataset = None
        self.val_dataset = None
        self._global_norm = None

    def setup(self, stage: str = None):
        """Instantiates the dataset and creates the 80/20 split."""
        full_dataset = EQDataset(self.data_dir)

        train_size = int(0.8 * len(full_dataset))
        val_size = len(full_dataset) - train_size

        # Get reproducible indices for the split (same seed as before)
        generator = torch.Generator().manual_seed(42)
        indices = torch.randperm(len(full_dataset), generator=generator).tolist()
        train_indices = indices[:train_size]
        val_indices = indices[train_size:]

        # Compute global normalization statistics from training data ONLY.
        # This prevents data leakage — val stats must not influence training.
        sample_indices = random.Random(42).sample(train_indices, min(500, len(train_indices)))
        channel_sums = np.zeros(3, dtype=np.float64)
        channel_sq_sums = np.zeros(3, dtype=np.float64)
        n_pixels = 0
        for idx in sample_indices:
            data = np.load(full_dataset.files[idx])
            if "mel_stack" in data:
                mel = data["mel_stack"].astype(np.float64)
            else:
                wet = data["wet_mel"].astype(np.float64)
                dry = data["dry_mel"].astype(np.float64)
                diff = wet - dry
                mel = np.stack([wet, dry, diff], axis=0)
            for c in range(3):
                channel_sums[c] += mel[c].sum()
                channel_sq_sums[c] += (mel[c] ** 2).sum()
            n_pixels += mel.shape[1] * mel.shape[2]

        means = channel_sums / (n_pixels * 1)
        stds = np.sqrt(channel_sq_sums / (n_pixels * 1) - means ** 2)
        stds = np.clip(stds, 1e-6, None)
        self._global_norm = (means.tolist(), stds.tolist())

        print(f"Global norm stats — mean: {means}, std: {stds}")

        # Create separate datasets: train with augmentation, val without
        train_full = EQDataset(self.data_dir, training_mode=True,
                               cache_in_ram=False, global_norm=self._global_norm)
        val_full = EQDataset(self.data_dir, training_mode=False,
                             cache_in_ram=False, global_norm=self._global_norm)

        self.train_dataset = Subset(train_full, train_indices)
        self.val_dataset = Subset(val_full, val_indices)

    def train_dataloader(self):
        return DataLoader(
            self.train_dataset,
            batch_size=self.batch_size,
            shuffle=True,
            num_workers=self.num_workers,
            pin_memory=True,
            persistent_workers=True if self.num_workers > 0 else False
        )

    def val_dataloader(self):
        return DataLoader(
            self.val_dataset,
            batch_size=self.batch_size,
            shuffle=False,
            num_workers=self.num_workers,
            pin_memory=True,
            persistent_workers=True if self.num_workers > 0 else False
        )
